Descrição

Retorna uma lista de updates de segurança coletados das ferramentas monitoradas (Veracode, SD Elements). Suporta filtragem por ferramenta e limite de resultados.

Query Parameters
tool string opcional Filtrar por ferramenta. Aceita regex (ex: Veracode, SD.*)
limit number opcional Limite de resultados. Padrão: 20. Use 0 para retornar todos.
Resposta de Sucesso
200 OK
Exemplos de Código
# Buscar últimos 20 updates
curl -X GET "https://afrikacrawlerapi.vercel.app/api/updates"

# Filtrar por Veracode, limite de 10
curl -X GET "https://afrikacrawlerapi.vercel.app/api/updates?tool=Veracode&limit=10"

# Buscar todos os updates
curl -X GET "https://afrikacrawlerapi.vercel.app/api/updates?limit=0"
// Buscar updates com filtros
async function fetchUpdates(tool = '', limit = 20) {
    const params = new URLSearchParams();
    if (tool) params.append('tool', tool);
    if (limit) params.append('limit', limit);
    
    const url = `https://afrikacrawlerapi.vercel.app/api/updates?${params}`;
    const response = await fetch(url);
    return await response.json();
}

// Exemplo de uso
const updates = await fetchUpdates('Veracode', 10);
console.log(`Total: ${updates.length} updates`);
import requests

def fetch_updates(tool=None, limit=20):
    """Busca updates da API Afrika Crawler."""
    params = {}
    if tool:
        params['tool'] = tool
    if limit:
        params['limit'] = limit
    
    response = requests.get(
        'https://afrikacrawlerapi.vercel.app/api/updates',
        params=params
    )
    return response.json()

# Exemplo de uso
updates = fetch_updates(tool='Veracode', limit=10)
print(f"Total: {len(updates)} updates")
